  <script>
    function saveQuestions() {
      const input = document.getElementById('question-input').value.trim();
      const lines = input.split('\n').map(line => line.trim()).filter(line => line !== "");

      const questions = [];
      let currentQuestion = null;
      let errorMsg = "";

      lines.forEach(line => {
        if (line.startsWith('/Q/')) {
          if (currentQuestion) {
            questions.push(currentQuestion);
          }
          currentQuestion = {
            question: line.replace('/Q/', '').trim(),
            correct: null,
            options: []
          };
        } else if (line.startsWith('/R/')) {
          if (currentQuestion) {
            currentQuestion.correct = line.replace('/R/', '').trim();
          }
        } else if (line.startsWith('/T/')) {
          if (currentQuestion) {
            currentQuestion.options.push(line.replace('/T/', '').trim());
          }
        }
      });

      if (currentQuestion) {
        questions.push(currentQuestion);
      }

      // Validação
      if (questions.length !== 10) {
        errorMsg = "Você deve cadastrar exatamente 10 perguntas (/Q/).";
      } else {
        for (let i = 0; i < questions.length; i++) {
          const q = questions[i];
          if (!q.question || !q.correct || q.options.length !== 3) {
            errorMsg = `Erro na pergunta ${i + 1}: ela deve ter 1 /R/ e 3 /T/.`;
            break;
          }
        }
      }

      if (errorMsg) {
        document.getElementById('error-msg').textContent = errorMsg;
        return;
      }

      // Monta estrutura final para o jogo
      const formatted = questions.map(q => {
        const allOptions = [...q.options];
        const correctIndex = Math.floor(Math.random() * 4);
        allOptions.splice(correctIndex, 0, q.correct);

        return {
          question: q.question,
          options: allOptions,
          correct: correctIndex
        };
      });

      // Salva no localStorage
      localStorage.setItem('customQuestions', JSON.stringify(formatted));

      // Volta para a tela inicial
      window.location.href = "index.html";
    }
  </script>
