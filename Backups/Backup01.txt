  <script>
    let questions = JSON.parse(localStorage.getItem('customQuestions')) || [
        {
            question: "Qual √© a capital do Brasil?",
            options: ["S√£o Paulo", "Rio de Janeiro", "Bras√≠lia", "Salvador"],
            correct: 2
        },
        {
            question: "Quanto √© 7 x 8?",
            options: ["54", "56", "64", "58"],
            correct: 1
        },
        {
            question: "Quem pintou a Mona Lisa?",
            options: ["Van Gogh", "Leonardo da Vinci", "Michelangelo", "Picasso"],
            correct: 1
        }
    ];

    let currentQuestion = 0;

    // Fun√ß√£o para iniciar o jogo
    function startGame() {
      // Esconde a tela inicial
      document.getElementById('home-screen').classList.add('hidden');
      
      // Mostra a tela do jogo
      const gameScreen = document.getElementById('game-screen');
      gameScreen.classList.remove('hidden');
      gameScreen.style.display = 'flex';

      // Carrega a primeira pergunta
      loadQuestion();
    }

    // Bot√£o de sair
    function exitGame() {
      alert("Fechando o jogo.");
    }

    // Carrega a pergunta atual
    function loadQuestion() {
      const q = questions[currentQuestion];
      document.getElementById('question-text').textContent = q.question;
      const optionsContainer = document.getElementById('options-container');
      const resultText = document.getElementById('result-text');
      resultText.textContent = "";
      optionsContainer.innerHTML = "";

      const letters = ['A', 'B', 'C', 'D'];

      q.options.forEach((option, index) => {
        const btn = document.createElement("div");
        btn.className = "option-button";
        btn.onclick = () => checkAnswer(index);

        const label = document.createElement("div");
        label.className = "option-label";
        label.textContent = letters[index];

        const text = document.createElement("span");
        text.textContent = option;

        btn.appendChild(label);
        btn.appendChild(text);
        optionsContainer.appendChild(btn);
      });
    }

    // Verifica a resposta
    function checkAnswer(index) {
      const q = questions[currentQuestion];
      const resultText = document.getElementById('result-text');

      if (index === q.correct) {
        resultText.textContent = "‚úÖ Resposta correta!";
      } else {
        resultText.textContent = "‚ùå Resposta errada!";
      }

      currentQuestion++;

      if (currentQuestion < questions.length) {
        setTimeout(loadQuestion, 2000);
      } else {
        setTimeout(() => {
          resultText.textContent = "üéâ Fim do jogo! Obrigado por jogar.";
          document.getElementById('options-container').innerHTML = "";
        }, 2000);
      }
    }
  </script>
